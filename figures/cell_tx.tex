\newcommand\chainArr[4]{
    \draw[very thick, blue, >=stealth,->] ({#1},{#2}) -- ({#1+0.5},{#2}) -- 
    ({#1+0.5},{0.5*#4+0.5*#2}) -- ({#3-0.5},{0.5*#4+0.5*#2}) -- 
    ({#3-0.5},{#4}) -- ({#3},{#4});
}
